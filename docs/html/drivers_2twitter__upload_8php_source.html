43  /* @brief Perform request to get media ID of image to be sent
44  *
45  * @return none
46  */
47  function getMediaId()
48  {
49  // Create new Twitter object
50  $twitter = new TwitterAPIExchange($this->apiKeys);
51 
52  // Get file contents of picture to Tweet
53  $picture = file_get_contents("birdcafe.png");
54  // Encode this data into correct format
55  $pictureData = base64_encode($picture);
56 
57  // First, get media ID from Twitter, then use that to upload image
58  $url = "https://upload.twitter.com/1.1/media/upload.json";
59  $requestMethod = "POST";
60  $setField = array("media_data" => $pictureData);
61 
62  // Make request
63  $pictureData = $this->twitter->setPostfields($setField)
64  ->buildOauth($url, $requestMethod)
65  ->performRequest();
66 
67  // Decode image data so it can be uploaded
68  $pictureData = @json_decode($pictureData, TRUE);
69  // Needs to be media_string_id, not media_string
70  $mediaID = $pictureData["media_id_string"];
71  $this->mediaID = $mediaID;
72  }
73 
74  /* @brief Use media ID to upload image to Twitter
75  *
76  * @return none
77  */
78  function uploadImage()
79  {
80  // Now that media_id for image has been obtained, upload image
81  $url = "https://api.twitter.com/1.1/statuses/update.json";
82  $requestMethod = "POST";
83  $setField = array(
84  "status" => "Test picture",
85  "possibly_sensitive" => FALSE,
86  "media_ids" => $this->mediaID);
87  // Make request for upload
88  $this->twitter->setPostfields($setField)
89  ->buildOauth($url, $requestMethod)
90  ->performRequest();
91  }
