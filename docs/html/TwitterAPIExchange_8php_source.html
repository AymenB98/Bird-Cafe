215  public function buildOauth($url, $requestMethod)
216  {
217  if (!in_array(strtolower($requestMethod), array('post', 'get', 'put', 'delete')))
218  {
219  throw new Exception('Request method must be either POST, GET or PUT or DELETE');
220  }
221

223  $consumer_secret = $this->consumer_secret;

225  $oauth_access_token_secret = $this->oauth_access_token_secret;
226 
227  $oauth = array(
228  'oauth_consumer_key' => $consumer_key,
229  'oauth_nonce' => time(),
230  'oauth_signature_method' => 'HMAC-SHA1',
231  'oauth_token' => $oauth_access_token,
232  'oauth_timestamp' => time(),
233  'oauth_version' => '1.0'
234  );
235 
236  $getfield = $this->getGetfield();
237

239  {
240  $getfields = str_replace('?', '', explode('&', $getfield));
241 
242  foreach ($getfields as $g)
243  {
244  $split = explode('=', $g);
245 
247  if (isset($split[1]))
248  {
249  $oauth[$split[0]] = urldecode($split[1]);
250  }
251  }
252  }
253 
254  $postfields = $this->getPostfields();
255 
256  if (!is_null($postfields)) {

259  }
