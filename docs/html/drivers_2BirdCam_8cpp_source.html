78 void BirdCam::takePhoto()
79 {
80  std::chrono::steady_clock::time_point photoTimerStart = std::chrono::steady_clock::now();
81 
82  //capture
83  raspicam::RaspiCam::grab();
84  //allocate memory
85  data=new unsigned char[raspicam::RaspiCam::getImageTypeSize ( raspicam::RASPICAM_FORMAT_RGB )];
86  //extract the image in rgb format
87  raspicam::RaspiCam::retrieve(data);//get camera image
88  //save
89  if (FilePath.empty()) //if no filepath set, default to birdcafe.ppm in local, print warning.
90  {
91  FilePath = "birdcafe.jpg";
92  std::cerr << "WARNING: No file path set, picture will be saved in local directory." <<std::endl << "Set a file path using BirdCam::setFilePath()." <<std::endl;
93  }
94  std::ofstream outFile(FilePath, std::ios::binary);
95  outFile<<"P6\n"<<raspicam::RaspiCam::getWidth() <<" "<<raspicam::RaspiCam::getHeight() <<" 255\n";
96  outFile.write ( ( char* ) data, raspicam::RaspiCam::getImageTypeSize ( raspicam::RASPICAM_FORMAT_RGB ) );
97  if (checkFilePathExists(FilePath) == false)
98  {
99  std::cerr << "ERROR: Filepath: " << FilePath << " does not exist." << std::endl << "Filepath has been set to /birdcafe.jpg" << std::endl;
100  FilePath = "birdcafe.jpg"; //If non-existant filepath given, default to local directory, named birdcafe.ppm
101  std::ofstream outFile(FilePath, std::ios::binary);
102  outFile<<"P6\n"<<raspicam::RaspiCam::getWidth() <<" "<<raspicam::RaspiCam::getHeight() <<" 255\n";
103  outFile.write ( ( char* ) data, raspicam::RaspiCam::getImageTypeSize ( raspicam::RASPICAM_FORMAT_RGB ) );
104  }
105  std::chrono::steady_clock::time_point photoTimerStop = std::chrono::steady_clock::now();
106  photoDuration = std::chrono::duration <float> (photoTimerStop - photoTimerStart).count();
107  fprintf(photoDurLog, "%f\n", 1000*photoDuration);
108 
109 }
